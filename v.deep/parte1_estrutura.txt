#!/bin/bash
set -euo pipefail

compute_type: "int8"
cpu_threads: 2
beam_size: 1
condition_on_previous_text: false
min_duration: 20
max_duration: 30
pause_threshold_ms: 300
protection_level: "basic"
EOF

echo "✅ Parte 1 concluída: estrutura criada em $PROJETO_DIR"
